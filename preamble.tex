}

    name={capture},
    description={Capture refers to taking a piece from the board, so your opponent is a piece down.\cite{unimelb:terms}}

    name={castling},
    description={Castling involves the king and the rook swapping positions. The king moves two spaces from the starting position to the left or right, and the rook moves to next to it on the other side.\cite{unimelb:terms}}

    name={en passant},
    description={A pawn that moves two squares forward can be taken by an opposing pawn that’s directly next to it on the following move.\cite{unimelb:terms}}
