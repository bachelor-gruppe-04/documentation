\makeglossaries

\newglossaryentry{check}{
    name={check}, 
    description={An attack on the king, but unlike checkmate, this is one that your opponent \textit{can} escape}
}

\newglossaryentry{checkmate}{
    name={checkmate}, 
    description={An attack on the King that your opponent can’t escape}
}

\newglossaryentry{stalemate}{
    name={stalemate}, 
    description={The player to move isn’t in check, but they can’t move any of their pieces. It’s a draw}
}

\newglossaryentry{capture}{
    name={capture}, 
    description={Taking a piece from the board, so your opponent is a piece down}
}

\newglossaryentry{castling}{
    name={castling}, 
    description={The king and the rook swapping positions. The king moves two spaces from the starting position to the left or right, and the rook moves to next to it on the other side}
}

\newglossaryentry{en-passant}{
    name={en passant}, 
    description={A pawn that moves two squares forward can be taken by an opposing pawn that’s directly next to it on the following move}
}

\newglossaryentry{promotion}{
    name={promotion}, 
    description={A pawn that reaches the end of the board can become any piece you want (not to a king or another pawn)}
}

\newglossaryentry{stockfish}{
    name={Stockfish}, 
    description={The strongest chess engine available to the public}
}

\newglossaryentry{lichess}{
    name={Lichess}, 
    description={A free and open-source Internet chess server}
}

\newglossaryentry{fide}{
    name={FIDE}, 
    description={The governing body of the sport of chess, and it regulates all international chess competitions.}
}

\newglossaryentry{elo}{
    name={Elo rating}, 
    description={A method for calculating the relative skill levels of players in zero-sum games such as chess}
}

\newglossaryentry{bullet}{
    name={bullet}, 
    description={Refers to a game less than three minutes per player, based on a 40-move game}
}

\newglossaryentry{blitz}{
    name={blitz}, 
    description={Refers to a game that has a fast time control. Base time per player 3 - 10 minutes}
}

\newglossaryentry{classical}{
    name={classical}, 
    description={Refers to a game with a total time of at least 65 minutes per player}
}

\newglossaryentry{back-rank}{
    name={back-rank}, 
    description={Refers to the first row (rank) of each player's side at the start of the game. In classical includes main (non-pawn) pieces, such as Rooks, Knights, Bishops, Queen, and King}
}

\newglossaryentry{chess960}{
    name={Chess960}, 
    description={Also known as Fischer Random Chess, is a chess variant that randomizes the starting position of the pieces on the back rank.}
}

\newglossaryentry{horde}{
    name={Horde}, 
    description={A chess variant where Black starts with the regular chess army and White with nothing but 36 pawns}
}

\newglossaryentry{racing-kings}{
    name={Racing Kings}, 
    description={A chess variant where the ultimate goal of each player is to take their kings to the eighth rank}
}
